window.onload = function() {
    var currencyMediumSelect = document.getElementById('currency-medium');
    var currencyMediumOptions = ['Dollar', 'EURo', 'Pound']; 

    for (var i = 0; i < currencyMediumOptions.length; i++) {
        var option = document.createElement('option');
        option.text = currencyMediumOptions[i];
        option.value = currencyMediumOptions[i];
        currencyMediumSelect.add(option);
    }
}

function convertToRsinSelectedMedium() {
    var amountRupees = parseFloat(document.getElementById('amount-rupees').value);
    var selectedMedium = document.getElementById('currency-medium').value;
    var convertedAmount = amountRupees * 2.2; 
    
    document.getElementById('converted-amount').value = convertedAmount + ' ' + selectedMedium;
}

function ConvertRs() {
    var selectmed2 = parseFloat(document.getElementById('amount-selected-medium').value);
    var previousMedium = document.getElementById('previous-medium').value;
   
    var convertedRupees = selectmed2 * 3.6; 
    
    document.getElementById('converted-rupees').value = convertedRupees + ' RS';
    document.getElementById('previous-medium').value = previousMedium;
}

function ResetFoo() {
    var selectedMedium = document.getElementById('currency-medium').value;
    
    document.getElementById('amount-rupees').value = '';
    document.getElementById('converted-amount').value = '';
    document.getElementById('amount-selected-medium').value = '';
    document.getElementById('converted-rupees').value = '';
    document.getElementById('previous-medium').value = selectedMedium;
    document.getElementById('programmed-by').value = ' Muhammad Fawwaz and 22F-8813';
}
